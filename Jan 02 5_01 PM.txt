        /* --- 2. SCREEN NAVIGATION --- */
        function goToScreen(screenNumber) {
            // Hide current active screens
            document.querySelectorAll('.screen').forEach(s => {
                s.classList.remove('active');
                setTimeout(() => {
                    if(!s.classList.contains('active')) s.classList.add('hidden');
                }, 800);
            });

            // Show new screen
            const nextScreen = document.getElementById(`screen-${screenNumber}`);
            nextScreen.classList.remove('hidden');
            // Small delay to allow 'hidden' class removal to take effect before opacity transition
            setTimeout(() => {
                nextScreen.classList.add('active');
            }, 50);

            // Special Logic for Screen 2 (Animations)
            if (screenNumber === 2) {
                playCompliments();
            }
        }

        }

        /* --- 5. CONFETTI ENGINE (Vanilla JS) --- */
        function startConfetti() {
            const canvas = document.getElementById('confetti-canvas');
            const ctx = canvas.getContext('2d');
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;

            const particles = [];
            const particleCount = 150;
            const colors = ['#ff8fa3', '#c9cbff', '#ffd700', '#64e7db'];

            for (let i = 0; i < particleCount; i++) {
                particles.push({
                    x: Math.random() * canvas.width,
                    y: Math.random() * canvas.height - canvas.height,
                    size: Math.random() * 8 + 4,
                    color: colors[Math.floor(Math.random() * colors.length)],
                    speedY: Math.random() * 3 + 2,
                    speedX: Math.random() * 2 - 1,
                    rotation: Math.random() * 360,
                    rotationSpeed: Math.random() * 5 - 2
                });
            }

            function animateConfetti() {
                ctx.clearRect(0, 0, canvas.width, canvas.height);
                
                particles.forEach(p => {
                    ctx.save();
                    ctx.fillStyle = p.color;
                    ctx.translate(p.x, p.y);
                    ctx.rotate((p.rotation * Math.PI) / 180);
                    ctx.fillRect(-p.size / 2, -p.size / 2, p.size, p.size);
                    ctx.restore();

                    p.y += p.speedY;
                    p.x += p.speedX;
                    p.rotation += p.rotationSpeed;

                    if (p.y > canvas.height) {
                        p.y = -20;
                        p.x = Math.random() * canvas.width;
                    }
                });

                requestAnimationFrame(animateConfetti);
            }

            animateConfetti();
        }
        
        // Handle Resize for Canvas

            const canvas = document.getElementById('confetti-canvas');
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        });
